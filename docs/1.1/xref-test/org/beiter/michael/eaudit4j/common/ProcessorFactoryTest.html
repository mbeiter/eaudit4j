98      /**
99       * Retrieve a specific implementation of the Processor interface, and assert that the returned implementation
100      * equals the requested implementation.
101      */
102     @Test
103     public void getSpecificImplementationTest() {
104 
105         CommonProperties commonProps = MapBasedCommonPropsBuilder.buildDefault();
106 
107         Processor processor;
108         try {
109             processor = ProcessorFactory.getInstance(CLASS_NAME, commonProps);
110         } catch (FactoryException e) {
111             AssertionError ae = new AssertionError("Instantiation error");
112             ae.initCause(e);
113             throw ae;
114         }
115 
116         String error = "The class instantiated by the factory does not match the expected class";
117         assertThat(error, ProcessorDummy.class.getCanonicalName(), is(equalTo(CLASS_NAME)));
118         assertThat(error, processor.getClass().getCanonicalName(), is(equalTo(CLASS_NAME)));
119     }

121     /**
122      * Retrieve two instances of a specific implementation of the Processor interface, and asserts that the returned
123      * objects are two separate instances.
124      */
125     @Test
126     public void twoInstancesAreDifferentTest() {
127 
128         CommonProperties commonProps = MapBasedCommonPropsBuilder.buildDefault();
129 
130         Processor processor1, processor2;
131         try {
132             processor1 = ProcessorFactory.getInstance(CLASS_NAME, commonProps);
133             processor2 = ProcessorFactory.getInstance(CLASS_NAME, commonProps);
134         } catch (FactoryException e) {
135             AssertionError ae = new AssertionError("Instantiation error");
136             ae.initCause(e);
137             throw ae;
138         }
139 
140         String error = "The factory returns a singleton instead of a new object";
141         assertThat(error, processor1, is(not(sameInstance(processor2))));
142     }
143 
144     /**
145      * Retrieve two singleton instances of a specific implementation of the Processor interface, and asserts that the
146      * two returned objects are identical (i.e. the factory returns a singleton).
147      * <p>
148      * Then, a regular (non-singleton) instance is retrieved, which are asserted to be different than the previously
149      * retrieved objects.
150      * <p>
151      * Finally, the factory is reset, and another instance is retrieved. If the factory resets properly, the third
152      * instance must be unequal to the first three instances.
153      */
154     @Test
155     public void factoryReturnsSingletonTest() {
156 
157         CommonProperties commonProps = MapBasedCommonPropsBuilder.buildDefault();
158 
159         // test that two singletons retrieved from the factory are identical
160         Processor processor1, processor2;
161         try {
162             processor1 = ProcessorFactory.getSingleton(CLASS_NAME, commonProps);
163             processor2 = ProcessorFactory.getSingleton(CLASS_NAME, commonProps);
164         } catch (FactoryException e) {
165             AssertionError ae = new AssertionError("Instantiation error");
166             ae.initCause(e);
167             throw ae;
168         }
169 
170         String error = "The factory does not return a singleton";
171         assertThat(error, processor1, is(sameInstance(processor2)));
172 
173         // then test that a regular (non-singleton) instance is different
174         Processor processor3;
175         try {
176             processor3 = ProcessorFactory.getInstance(CLASS_NAME, commonProps);
177         } catch (FactoryException e) {
178             AssertionError ae = new AssertionError("Instantiation error");
179             ae.initCause(e);
180             throw ae;
181         }
182         error = "The factory returns a singleton instead of a new object";
183         assertThat(error, processor1, is(not(sameInstance(processor3))));
184         assertThat(error, processor2, is(not(sameInstance(processor3))));
185 
186         // reset the factory
187         ProcessorFactory.reset();
188 
189         // now test that the factory return a new object (i.e. a new singleton)
190         Processor processor4;
191         try {
192             processor4 = ProcessorFactory.getSingleton(CLASS_NAME, commonProps);
193         } catch (FactoryException e) {
194             AssertionError ae = new AssertionError("Instantiation error");
195             ae.initCause(e);
196             throw ae;
197         }
198 
199         error = "The factory does not return a singleton, or does not reset properly";
200         assertThat(error, processor1, is(not(sameInstance(processor4))));
201         assertThat(error, processor2, is(not(sameInstance(processor4))));
202         assertThat(error, processor3, is(not(sameInstance(processor4))));
203     }
