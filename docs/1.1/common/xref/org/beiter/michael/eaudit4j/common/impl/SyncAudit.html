33  package org.beiter.michael.eaudit4j.common.impl;
34  
35  import org.apache.commons.lang3.StringUtils;
36  import org.apache.commons.lang3.Validate;
37  import org.beiter.michael.eaudit4j.common.Audit;
38  import org.beiter.michael.eaudit4j.common.AuditErrorConditions;
39  import org.beiter.michael.eaudit4j.common.AuditException;
40  import org.beiter.michael.eaudit4j.common.CommonProperties;
41  import org.beiter.michael.eaudit4j.common.Event;
42  import org.beiter.michael.eaudit4j.common.FactoryException;
43  import org.beiter.michael.eaudit4j.common.ProcessingObjects;
44  import org.beiter.michael.eaudit4j.common.Processor;
45  import org.beiter.michael.eaudit4j.common.ProcessorFactory;
46  import org.beiter.michael.eaudit4j.common.propsbuilder.MapBasedCommonPropsBuilder;
47  import org.slf4j.Logger;
48  import org.slf4j.LoggerFactory;
49  
50  import java.util.ArrayList;
51  import java.util.Collections;
52  import java.util.List;
53  
54  /**
55   * This class provides a synchronous implementation of {@link Audit}.
56   */
57  public class SyncAudit
58          implements Audit {
59  
60      /**
61       * The logger object for this class
62       */
63      private static final Logger LOG = LoggerFactory.getLogger(SyncAudit.class);
64  
65      /**
66       * A copy of the common properties, initialize with a default config set:
67       */
68      private CommonProperties commonProps = MapBasedCommonPropsBuilder.buildDefault();
69  
70      /**
71       * The List itself is not thread safe, but it also does not need to be modified in this class after initialization
72       * (i.e. build once in {@code init()), and then read only from there on). Repeated calls to {@code init()} will
73       * create a new list, and then replace this list reference.
74       */
75      private List<Processor> processors = new ArrayList<>();
76  
77      /**
78       * {@inheritDoc}
79       */
80      @Override
81      public final void init(final CommonProperties properties)
82              throws AuditException {
83  
84          // store a local copy of the properties
85          this.commonProps = new CommonProperties(properties);
86  
87          // get the config string will the processors from the configuration
88          final String processorClasses = this.commonProps.getProcessors();
89          if (processorClasses != null && processorClasses.length() > 0) {
90  
91              // split the processor config string to give us a list of classes to instantiate
92              final String[] procClassesList = processorClasses.split(",");
93  
94              // we want to be thread safe with the configuration, hence we create a local instance of the audit processor
95              // chain, and assign it at the end of the init() operation in a single call
96              final List<Processor> tmpProcessors = new ArrayList<>(procClassesList.length);
97  
98              // the processors are stored from left to right in the config file, so we can simply create them and add
99              // them to the list of registered processors (which we will invoke later on in the audit() methods) in the
100             // provided order
101             for (final String procClassName : procClassesList) {
102 
103                 if (procClassName != null && !procClassName.isEmpty()) {
104 
105                     Processor processor;
106                     try {
107                         processor = ProcessorFactory.getInstance(procClassName, this.commonProps);
108                     } catch (FactoryException e) {
109                         final String error = "Failed to retrieve instance of processor class: " + procClassName;
110                         LOG.warn(error, e);
111                         throw new AuditException(AuditErrorConditions.INITIALIZATION, error, e);
112                     }
113 
114                     // add the newly created processor to the local tmpProcessors list
115                     tmpProcessors.add(processor);
116                 }
117             }
118 
119             // swap the processors reference, and make sure that the list cannot be accidentally modified
120             processors = Collections.unmodifiableList(tmpProcessors);
121         }
122     }
123 
124     /**
125      * {@inheritDoc}
126      */
127     @Override
128     public final Event audit(final Event event)
129             throws AuditException {
130 
131         if (event == null) {
132             final String error = "The validated object 'event' is null";
133             LOG.warn(error);
134             throw new AuditException(AuditErrorConditions.INVALID_EVENT, error);
135         }
136 
137         final String auditStream = commonProps.getDefaultAuditStream();
138         if (auditStream == null || StringUtils.isBlank(auditStream)) {
139             final String error = "The validated character sequence 'auditStream' is null or empty";
140             LOG.warn(error);
141             throw new AuditException(AuditErrorConditions.CONFIGURATION, error);
142         }
143 
144         return audit(event, auditStream);
145     }
146 
147     /**
148      * {@inheritDoc}
149      */
150     @Override
151     public final Event audit(final Event event, final String auditStreamName)
152             throws AuditException {
153 
154         if (event == null) {
155             final String error = "The validated object 'event' is null";
156             LOG.warn(error);
157             throw new AuditException(AuditErrorConditions.INVALID_EVENT, error);
158         }
159 
160         Validate.notBlank(auditStreamName, "The validated character sequence 'auditStreamName' is null or empty");
161 
162         // We create the set of default (empty) ProcessingObjects every time this method is invoked to prevent
163         // interference, because there is zero protection on that object by design (processors can modify the objects
164         // list and the objects themselves at will, there are no defensive copies or other isolation).
165         final ProcessingObjects processingObjects = new ProcessingObjects();
166 
167         return audit(event, auditStreamName, processingObjects);
168     }
169 
170     /**
171      * {@inheritDoc}
172      */
173     // suppress warnings about using a StringBuffer for the error concatenation (only used for exception handling)
174     @SuppressWarnings("PMD.UseStringBufferForStringAppends")
175     @Override
176     public final Event audit(final Event event, final String auditStreamName, final ProcessingObjects processingObjects)
177             throws AuditException {
178 
179         if (event == null) {
180             final String error = "The validated object 'event' is null";
181             LOG.warn(error);
182             throw new AuditException(AuditErrorConditions.INVALID_EVENT, error);
183         }
184 
185         Validate.notBlank(auditStreamName, "The validated character sequence 'auditStreamName' is null or empty");
186         Validate.notNull(processingObjects, "The validated object 'processingObjects' is null");
187 
188         // if the array with processors is empty, no processors have been configured.
189         // Log an error (and fail) or just log a warning, depending on the configuration:
190         if (processors == null || processors.isEmpty()) {
191 
192             String error = "No processors configured for auditing subsystem";
193             if (commonProps.isFailOnMissingProcessors()) {
194                 LOG.warn(error);
195                 throw new AuditException(AuditErrorConditions.CONFIGURATION, error);
196             } else {
197                 error += ". Audit events are neither being processed, nor persisted.";
198                 LOG.warn(error);
199                 return event;
200             }
201         } else {
202 
203             // Go through the processor chain, have each processor work on the event,
204             // and feed the modified event into the next processor down the chain:
205             Event tmpEvent = event; // start the chain with the original event
206             for (final Processor processor : processors) {
207 
208                 tmpEvent = processor.process(tmpEvent, auditStreamName, processingObjects);
209             }
210 
211             // once all processing is complete, return the (potentially modified) Event
212             return tmpEvent;
213         }
214     }
