106     /**
107      * Creates a set of Cassandra properties from an existing set of Cassandra properties, making a defensive copy.

114     public CassandraProperties(final CassandraProperties properties) {
115         this();
116 
117         Validate.notNull(properties, "The validated object 'properties' is null");
118 
119         setInsertEventCqlStmt(properties.getInsertEventCqlStmt());
120         setEventIdCqlParam(properties.getEventIdCqlParam());
121         setAuditStreamNameCqlParam(properties.getAuditStreamNameCqlParam());
122         setEventJsonCqlParam(properties.getEventJsonCqlParam());
123         setStringEncoding(properties.getStringEncoding());

125         setSessionName(properties.getSessionName());

127     }

347     /**
348      * @return Any additional properties stored in this object that have not explicitly been parsed
349      * @see CassandraProperties#setAdditionalProperties(Map)
350      */
351     public final Map<String, String> getAdditionalProperties() {
352 
353         // create a defensive copy of the map and all its properties
354         if (this.additionalProperties == null) {
355             // this should never happen!
356             return new ConcurrentHashMap<>();
357         } else {
358             final Map<String, String> tempMap = new ConcurrentHashMap<>();
359             tempMap.putAll(additionalProperties);
360 
361             return tempMap;
362         }
363     }

365     /**
366      * Any additional properties which have not been parsed, and for which no getter/setter exists, but are to be
367      * stored in this object nevertheless.
368      * <p>
369      * This property is commonly used to preserve original properties from upstream components that are to be passed
370      * on to downstream components unchanged. This properties set may or may not include properties that have been
371      * extracted from the map, and been made available through this POJO.
372      * <p>
373      * Note that these additional properties may be <code>null</code> or empty, even in a fully populated POJO where
374      * other properties commonly have values assigned to.
375      *
376      * @param additionalProperties The additional properties to store
377      */
378     public final void setAdditionalProperties(final Map<String, String> additionalProperties) {
379 
380         // create a defensive copy of the map and all its properties
381         if (additionalProperties == null) {
382             this.additionalProperties = new ConcurrentHashMap<>();
383         } else {
384             this.additionalProperties = new ConcurrentHashMap<>();
385             this.additionalProperties.putAll(additionalProperties);
386         }
387     }
