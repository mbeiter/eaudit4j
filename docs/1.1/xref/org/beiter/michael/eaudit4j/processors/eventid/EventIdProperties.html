35  package org.beiter.michael.eaudit4j.processors.eventid;
36  
37  import org.apache.commons.lang3.Validate;
38  
39  import java.util.Map;
40  import java.util.concurrent.ConcurrentHashMap;
41  
42  /**
43   * This class specifies properties specific to the Event ID Processor.
44   */
45  // suppress warnings about the long variable names
46  @SuppressWarnings("PMD.LongVariable")
47  public class EventIdProperties {
48  
49      /**
50       * @see EventIdProperties#setLength(int)
51       */
52      private int length;
53  
54      /**
55       * @see EventIdProperties#setEventFieldName(String)
56       */
57      private String eventFieldName;
58  
59      /**
60       * @see EventIdProperties#setAdditionalProperties(Map)
61       */
62      private Map<String, String> additionalProperties = new ConcurrentHashMap<>();
63  
64      /**
65       * Constructs an empty set of event ID properties, with most values being set to <code>null</code>, 0, or empty
66       * (depending on the type of the property). Usually this constructor is used if this configuration POJO is populated
67       * in an automated fashion (e.g. injection). If you need to build them manually (possibly with defaults), use or
68       * create a properties builder.
69       * <p>
70       * You can change the defaults with the setters.
71       *
72       * @see org.beiter.michael.eaudit4j.common.propsbuilder.MapBasedCommonPropsBuilder#buildDefault()
73       * @see org.beiter.michael.eaudit4j.common.propsbuilder.MapBasedCommonPropsBuilder#build(Map)
74       */
75      public EventIdProperties() {
76  
77          // no code here, constructor just for java docs
78      }
79  
80      /**
81       * Creates a set of event ID properties from an existing set of event ID properties, making a defensive copy.
82       *
83       * @param properties The set of properties to copy
84       * @throws NullPointerException When {@code properties} is {@code null}
85       * @see EventIdProperties ()
86       */
87      public EventIdProperties(final EventIdProperties properties) {
88  
89          this();
90  
91          Validate.notNull(properties, "The validated object 'properties' is null");
92  
93          setLength(properties.getLength());
94          setEventFieldName(properties.getEventFieldName());
95          setAdditionalProperties(properties.getAdditionalProperties());
96      }
97  
98      /**
99       * @return the length of the (random) event ID generated by this processor
100      * @see EventIdProperties#setLength(int)
101      */
102     public final int getLength() {
103 
104         // no need for defensive copies of int
105 
106         return length;
107     }
108 
109     /**
110      * The length in chars of the (random) event ID that is generated by this processor.
111      * <p>
112      * The entropy of this random String is approximately 3/4 of the overall String length.
113      *
114      * @param length the length of the (random) event ID generated by this processor (must be greater 0)
115      * @throws IllegalArgumentException When the provided value of {@code eventIdLength} is out of range
116      */
117     public final void setLength(final int length) {
118 
119         Validate.inclusiveBetween(1, Integer.MAX_VALUE, length);
120 
121         // no need for defensive copies of int
122 
123         this.length = length;
124     }
125 
126     /**
127      * @return The field name used to store the event ID in events
128      * @see EventIdProperties#setEventFieldName(String)
129      */
130     public final String getEventFieldName() {
131 
132         // no need for defensive copies of String
133 
134         return eventFieldName;
135     }
136 
137     /**
138      * Set the field name to be used when storing the event ID in audit events
139      *
140      * @param eventFieldName The field name to store the event ID in
141      * @throws NullPointerException When the {@code eventFieldName} is {@code null}
142      * @throws IllegalArgumentException When the {@code eventFieldName} is {@code empty}
143      */
144     public final void setEventFieldName(final String eventFieldName) {
145 
146         Validate.notBlank(eventFieldName, "The validated character sequence 'eventFieldName' is null or empty");
147 
148         // no need for defensive copies of String
149 
150         this.eventFieldName = eventFieldName;
151     }
152 
153     /**
154      * @return Any additional properties stored in this object that have not explicitly been parsed
155      * @see EventIdProperties#setAdditionalProperties(Map)
156      */
157     public final Map<String, String> getAdditionalProperties() {
158 
159         // create a defensive copy of the map and all its properties
160         if (this.additionalProperties == null) {
161             // this should never happen!
162             return new ConcurrentHashMap<>();
163         } else {
164             final Map<String, String> tempMap = new ConcurrentHashMap<>();
165             tempMap.putAll(additionalProperties);
166 
167             return tempMap;
168         }
169     }
170 
171     /**
172      * Any additional properties which have not been parsed, and for which no getter/setter exists, but are to be
173      * stored in this object nevertheless.
174      * <p>
175      * This property is commonly used to preserve original properties from upstream components that are to be passed
176      * on to downstream components unchanged. This properties set may or may not include properties that have been
177      * extracted from the map, and been made available through this POJO.
178      * <p>
179      * Note that these additional properties may be <code>null</code> or empty, even in a fully populated POJO where
180      * other properties commonly have values assigned to.
181      *
182      * @param additionalProperties The additional properties to store
183      */
184     public final void setAdditionalProperties(final Map<String, String> additionalProperties) {
185 
186         // create a defensive copy of the map and all its properties
187         if (additionalProperties == null) {
188             this.additionalProperties = new ConcurrentHashMap<>();
189         } else {
190             this.additionalProperties = new ConcurrentHashMap<>();
191             this.additionalProperties.putAll(additionalProperties);
192         }
193     }
194 }
